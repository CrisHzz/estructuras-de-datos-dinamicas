//Esta es mi propuesta para agregar un nodo a una lista ligada
public class Nodo {
    private Nodo proximo;
    private int valor;
    public Nodo(Nodo proximo, int valor) {
        this.proximo = proximo;
        this.valor = valor;
    }
    public static void main(String[] args) {
        Nodo n1 = new Nodo(null, 4);
        Nodo n2 = new Nodo(null, 3);
        Nodo n3 = new Nodo(null, 10);
        Nodo l1 = AgregarNodo(null, n1, null);
        Nodo l2 = AgregarNodo(l1, n2, null);
        Nodo l3 = AgregarNodo(l2, n3, null);

        //Asi imprimo en pantalla la lista ligada
        Nodo actual = l3;
        while (actual != null) {
            System.out.println(actual.getValor() + " ");
            actual = actual.getProximo();
        }
    }
    public Nodo getProximo() {
        return proximo;
    }
    public void setProximo(Nodo proximo) {
        this.proximo = proximo;
    }
    public int getValor() {
        return valor;
    }
    public void setValor(int valor) {
        this.valor = valor;
    }

	//Esta es la funcion AgregarNodo
    public static Nodo AgregarNodo(Nodo cabeza, Nodo ingresar, Nodo anterior) {
        if (cabeza == null) {
            if (anterior != null) {
                anterior.setProximo(ingresar);
            }
            return ingresar;
        }
        if (ingresar == null) {
            System.out.println("Â¡ERROR!");
            return cabeza;
        }

        if (ingresar.getValor() < cabeza.getValor()) {
            ingresar.setProximo(cabeza);
            if (anterior != null) {
                anterior.setProximo(ingresar);
            }
            return ingresar;
        }

        cabeza.setProximo(AgregarNodo(cabeza.getProximo(), ingresar, cabeza));
        return cabeza;
    }
}
